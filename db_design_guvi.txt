Sample DataBase model Comments for GUVI Zen class 
-------------------------------------------------

studentinfo {
	studentid int pk increments
	studentname varchar(255)
	mobilenumber varchar(255) unique
	email varchar(255) unique
	mentorname varchar(255)
	batchid int > batchinfo.batchid
	batchname varchar(255)
}

taskinfo {
	taskid int pk increments
	taskname varchar(255) unique
	studentid int > studentinfo.studentid
	mentorid int > mentorinfo.mentorid
	mentorname varchar(25)
	batchname varchar(50)
}

mentorinfo {
	mentorid int pk increments
	email varchar(50) unique
	contactno varchar(15) unique
	assignedbatch varchar(100)
}

batchinfo {
	batchid int pk increments
	batchname varchar(255)
	totalstudents int
	mentorid int > mentorinfo.mentorid
}
